60# Function: initialize_display
61# Description: Initializes the OLED display via I2C.
62# Returns: The initialized OLED object.
63# Raises: Exception if the I2C or OLED initialization fails.
64def initialize_display():
65 """
66 Set up the I2C bus and OLED display.
67 """
68 try:
69 # Initialize I2C bus
70 i2c = I2C(0, scl=Pin(I2C_SCL_PIN), sda=Pin(I2C_SDA_PIN), freq=I2C_FREQ)
71 # Initialize OLED display
72 oled = SH1106_I2C(i2c)
73 oled.contrast(OLED_CONTRAST)
74 return oled
75 except Exception as e:
76 print(f"Error initializing OLED: {e}")
77 raise

110# Function: parse_gps_data
111# Description: Parses raw GPS data from the MicropyGPS object into a dictionary.
112# Parameters:
113# my_gps (MicropyGPS): The MicropyGPS object handling GPS data parsing.

115def parse_gps_data(my_gps):
116 """
117 Extracts GPS data from the MicropyGPS object and formats it.
118 """
119 return {
120 'time': my_gps.timestamp, # Format time as HH:MM:SS
121 'date': my_gps.date_string('s_dmy'), # Date in DD/MM/YYYY format
122 'latitude': my_gps.latitude_string(), # Latitude in degrees/minutes/seconds
123 'longitude': my_gps.longitude_string(), # Longitude in degrees/minutes/seconds
124 'altitude': str(my_gps.altitude), # Altitude in meters
125 'satellites': str(my_gps.satellites_in_use), # Number of satellites in use
126 'hdop': str(my_gps.hdop), # Horizontal Dilution of Precision
127 }
