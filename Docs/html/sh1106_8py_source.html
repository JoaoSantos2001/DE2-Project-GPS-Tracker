23class SH1106_I2C(framebuf.FrameBuffer):
24 DEV_ADDR = 0x3c
25 WIDTH = 128
26 HEIGHT = 64
27 PAGES = HEIGHT // 8
28 LOW_COLUMN_ADDR = 0x00
29 HIGH_COLUMN_ADDR = 0x10
30 PAGE_ADDRESS = 0xb0

32 def __init__(self, i2c, width=WIDTH, height=HEIGHT, addr=DEV_ADDR):
33 self.i2c = i2c
34 self.addr = addr
35 self._sh1106_init()
36 self.buffer = bytearray(self.PAGES * self.WIDTH)
37 super().__init__(self.buffer, width, height, framebuf.MONO_VLSB)

39 def write_cmd(self, cmd):
40 """Write a byte of command to SH1106"""
41 self.i2c.writeto(self.DEV_ADDR, bytearray([0x80, cmd]))
42
43 def write_data(self, data):
44 """Write a databuffer to SH1106"""
45 self.i2c.writeto(self.DEV_ADDR, b"\x40"+data)
46
47 def poweron(self):
48 self.write_cmd(0xaf)
49
50 def poweroff(self):
51 self.write_cmd(0xae)
52
53 def sleep(self, value):
54 self.write_cmd(0xae | (not value))
55
56 def contrast(self, val):
57 self.write_cmd(0x81)
58 self.write_cmd(val)
59
60 def show(self):
61 (w, p, buf) = (self.WIDTH, self.PAGES, self.buffer)
62 for page in range(0, p):
63 self.write_cmd(self.PAGE_ADDRESS | page)
64 self.write_cmd(self.LOW_COLUMN_ADDR | 2)
65 self.write_cmd(self.HIGH_COLUMN_ADDR | 0)
66 # print(f"Updating page {page}")
67 self.write_data(buf[(w*page):(w*page+w)])

69 def _sh1106_init(self):
70 """Initialize SH1106"""
71 INIT_SEQ = (
72 0xae, # Turn off oled panel
73 0x00, # Set low column address
74 0x10, # Set high column address
75 0x40, # Set start line address
76 0x20, 0x02, # Page addressing mode
77 0xc8, # Top-down segment (4th segment)
78 0x81, # Set contrast control register
79 0xcf, 0xa1, # Set segment re-map 95 to 0
80 0xa6, # Set normal display
81 0xa8, # Set multiplex ratio(1 to 64)
82 0x3f, # 1/64 duty
83 0xd3, 0x00, # Set display offset: none
84 0xd5,
85 0x80,
86 0xd9,
87 0xf1, 0xda,
88 0x12, 0xdb,
89 0x40, 0x8d,
90 0x14,
91 0xaf # Turn on oled panel
92 )
93 time.sleep_ms(100)
94
95 for cmd in INIT_SEQ:
96 self.write_cmd(cmd)
