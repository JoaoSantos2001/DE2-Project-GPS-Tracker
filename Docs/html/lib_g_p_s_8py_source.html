448 def gpgsv(self):
449 """Parse Satellites in View (GSV) sentence. Updates number of SV Sentences,the number of the last SV sentence
450 parsed, and data on each satellite present in the sentence"""
451 try:
452 num_sv_sentences = int(self.gps_segments[1])
453 current_sv_sentence = int(self.gps_segments[2])
454 sats_in_view = int(self.gps_segments[3])
455 except ValueError:
456 return False
457
458 # Create a blank dict to store all the satellite data from this sentence in:
459 # satellite PRN is key, tuple containing telemetry is value
460 satellite_dict = dict()
461
462 # Calculate Number of Satelites to pull data for and thus how many segment positions to read
463 if num_sv_sentences == current_sv_sentence:
464 # Last sentence may have 1-4 satellites; 5 - 20 positions
465 sat_segment_limit = (sats_in_view - ((num_sv_sentences - 1) * 4)) * 5
466 else:
467 sat_segment_limit = 20 # Non-last sentences have 4 satellites and thus read up to position 20
468
469 # Try to recover data for up to 4 satellites in sentence
470 for sats in range(4, sat_segment_limit, 4):
471
472 # If a PRN is present, grab satellite data
473 if self.gps_segments[sats]:
474 try:
475 sat_id = int(self.gps_segments[sats])
476 except (ValueError,IndexError):
477 return False
478
479 try: # elevation can be null (no value) when not tracking
480 elevation = int(self.gps_segments[sats+1])
481 except (ValueError,IndexError):
482 elevation = None
483
484 try: # azimuth can be null (no value) when not tracking
485 azimuth = int(self.gps_segments[sats+2])
486 except (ValueError,IndexError):
487 azimuth = None
488
489 try: # SNR can be null (no value) when not tracking
490 snr = int(self.gps_segments[sats+3])
491 except (ValueError,IndexError):
492 snr = None
493 # If no PRN is found, then the sentence has no more satellites to read
494 else:
495 break
496
497 # Add Satellite Data to Sentence Dict
498 satellite_dict[sat_id] = (elevation, azimuth, snr)
499
500 # Update Object Data
501 self.total_sv_sentences = num_sv_sentences
502 self.last_sv_sentence = current_sv_sentence
503 self.satellites_in_view = sats_in_view
504
505 # For a new set of sentences, we either clear out the existing sat data or
506 # update it as additional SV sentences are parsed
507 if current_sv_sentence == 1:
508 self.satellite_data = satellite_dict
509 else:
510 self.satellite_data.update(satellite_dict)
511
512 return True

525 def update(self, new_char):
526 """Process a new input char and updates GPS object if necessary based on special characters ('$', ',', '*')
527 Function builds a list of received string that are validate by CRC prior to parsing by the appropriate
528 sentence function. Returns sentence type on successful parse, None otherwise"""
529
530 valid_sentence = False
531
532 # Validate new_char is a printable char
533 ascii_char = ord(new_char)
534
535 if 10 <= ascii_char <= 126:
536 self.char_count += 1
537
538 # Write Character to log file if enabled
539 if self.log_en:
540 self.write_log(new_char)
541
542 # Check if a new string is starting ($)
543 if new_char == '$':
544 self.new_sentence()
545 return None
546
547 elif self.sentence_active:
548
549 # Check if sentence is ending (*)
550 if new_char == '*':
551 self.process_crc = False
552 self.active_segment += 1
553 self.gps_segments.append('')
554 return None
555
556 # Check if a section is ended (,), Create a new substring to feed
557 # characters to
558 elif new_char == ',':
559 self.active_segment += 1
560 self.gps_segments.append('')
561
562 # Store All Other printable character and check CRC when ready
563 else:
564 self.gps_segments[self.active_segment] += new_char
565
566 # When CRC input is disabled, sentence is nearly complete
567 if not self.process_crc:
568
569 if len(self.gps_segments[self.active_segment]) == 2:
570 try:
571 final_crc = int(self.gps_segments[self.active_segment], 16)
572 if self.crc_xor == final_crc:
573 valid_sentence = True
574 else:
575 self.crc_fails += 1
576 except ValueError:
577 pass # CRC Value was deformed and could not have been correct
578
579 # Update CRC
580 if self.process_crc:
581 self.crc_xor ^= ascii_char
582
583 # If a Valid Sentence Was received and it's a supported sentence, then parse it!!
584 if valid_sentence:
585 self.clean_sentences += 1 # Increment clean sentences received
586 self.sentence_active = False # Clear Active Processing Flag
587
588 if self.gps_segments[0] in self.supported_sentences:
589
590 # parse the Sentence Based on the message type, return True if parse is clean
591 if self.supported_sentences[self.gps_segments[0]](self):
592
593 # Let host know that the GPS object was updated by returning parsed sentence type
594 self.parsed_sentences += 1
595 return self.gps_segments[0]
596
597 # Check that the sentence buffer isn't filling up with Garage waiting for the sentence to complete
598 if self.char_count > self.SENTENCE_LIMIT:
599 self.sentence_active = False
600
601 # Tell Host no new sentence was parsed
602 return None
